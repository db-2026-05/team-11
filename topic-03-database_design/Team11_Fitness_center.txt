//Members module for managing clients
Enum membership_type {
  monthly
  yearly
  premium
}

Table "members" {
  "member_id" SERIAL [pk, increment]
  "first_name" VARCHAR(100) [not null]
  "last_name" VARCHAR(100) [not null]
  "phone_number" VARCHAR(30) [unique]
  "birth_date" DATE
  "join_date" DATE [not null, default: 'CURRENT_DATE']
  "membership_type" membership_type [not null]
  "membership_start_date" DATE [not null]
  "membership_end_date" DATE [not null, note: 'CHECK (membership_end_date > membership_start_date)']
}

  "name" VARCHAR(50) [not null]
}

  monday
  tuesday
  wednesday
  thursday
  friday
  saturday
  sunday
}

Table "trainer_schedules" {
  "trainer_schedule_id" SERIAL [pk, increment]

  "weekday" weekday_enum [not null]

}

  "trainer_id" SERIAL [pk, increment]

  "phone_number" VARCHAR(30) [unique]
  "employed_date" DATE [not null]
  "experience_years" INT [not null]
  "is_active" BOOLEAN [not null, default: true]
}
